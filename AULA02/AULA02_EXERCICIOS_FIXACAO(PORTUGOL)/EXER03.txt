/* Ler um número inteiro e escrever o valor do seu quadrado e da 
sua raiz quadrada. Obs:utilizar a classe "math" e método "sqrt". 
ex.: math.sqrt(x); */ 

VAR
n1, quadrado, raizQuadrada: inteiro;

INÍCIO

ler n1;
quadrado <- n1*n1;
raizQuadrada <- raiz(n1);

escrever "Quadrado: "+quadrado+" Raiz quadrada: "+raizQuadrada+";";

FIM
