/* Ler o salário de um funcionário e imprimí-lo com um
   desconto de INSS de 14%;
 */
 
 VAR
 salario, desconto, salarioReajustado: real;
 
 INICIO
 
 ler salario;
 desconto <- (salario *(14/100));
 salarioReajustado <- (salario - desconto);
 
 escrever "Seu salario atual é :"+salario+"\n"+"Desconto INSS: "+desconto+"\n"+"Seu salario reajustado: " +
salarioReajustado; 

FIM
