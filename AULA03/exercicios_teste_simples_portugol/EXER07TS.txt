PROGRAMA

INICIO
vogal: caractere;

escreva(Descubra se é vogal, Obs:entrar com letras maiúsculas)
leia(vogal)

padrao(vogal)
{

		escreva(É VOGAL)
		pare
	caso "E":
		escreva(É VOGAL)
		pare
	caso "I":
		escreva(É VOGAL)
		pare
	caso "O":
		escreva(É VOGAL)
		pare
	caso "U":
		escreva(É VOGAL)
		pare
	caso contrario:
