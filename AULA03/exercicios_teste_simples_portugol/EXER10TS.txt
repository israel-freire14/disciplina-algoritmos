/*Ler o preço de duas blusas e imprimir qual é o maior.
Desconsidere empates. */

PROGRAMA


INICIO
precoBlusaA, precoBlusaB: Real;

escreva(Informe o valor da blusa A:)
leia(precoBlusaA)
escreva(Informe o valor da blusa B:)
leia(precoBlusaB)


se(precoBlusaA > precoBlusaB)
{
	escreva(Blusa A tem o maior preço)
}
senao se(precoBlusaB > precoBlusaA)
{
	escreva(Blusa B tem maior preço)
}


FIM
